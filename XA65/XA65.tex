  	  		    \begin{question} 
  	  		    True/False: If $\hat{x}$ is a least-squares solution of $A\vec{x} = \vec{b}$, then $A\hat{x}$ is a point in the column space of $A$ closest to $\vec{b}$.\\
  	  		    \begin{multipleChoice}
  	  		    	\choice[correct]{True}
  	  		    	\choice{False}
  	  		    	\end{multipleChoice}
  	  		    
  	  		    	\end{question}
  	  		    	%----------------------------------------------------------------
  	  		       \begin{question} 
  	  		    	When does $A\vec{x} = \vec{b}$ have exactly one least-squares solution? Select all that apply.
  	  		    	\begin{selectAll}
  	  		    		\choice{When $A$ is $n\times n$.}
  	  		    		\choice{When $A\vec{x} = \vec{b}$ is consistent.}
  	  		    		\choice[correct]{When the columns of $A$ are linearly independent.}
  	  		    		\choice[correct]{When $A\vec{x} = \vec{b}$ is has exactly one solution.}
  	  		    	\end{selectAll}
  	  		    	
  	  		    \end{question}
  	  		                      %----------------------------------------------------------------
  	  		                          \begin{question} 
  	  		                      Let $A = \begin{bmatrix} 4 &1\\0&2\\0&1\end{bmatrix}$ and $\vec{b}= \begin{bmatrix}1\\-1\\2\end{bmatrix}$.\\\\
  	  		                      \vspace{10pt}\\
  	  		                      Compute $A^TA$. $A^TA =\begin{bmatrix} \answer{16}& \answer{4}\\\answer{4}& \answer{6}\end{bmatrix}$\vspace{20pt}\\
  	  		                       Compute $A^T\vec{b}$. $A^T\vec{b} = \begin{bmatrix} \answer{4}\\\answer{1}\end{bmatrix}$\vspace{20pt}\\
  	  		                      Solve the normal equations $A^TA\vec{x} = A^T\vec{b}$ to determine the set of least-squares solutions.\\
  	  		                      $$\vec{x} = \begin{bmatrix}1/4\\ \answer{0}\end{bmatrix}$$
  	  		                      
  	  		                      Why is there only one least-squares solution?
  	  		                      
  	  		                      \begin{multipleChoice}
  	  		                      	\choice[correct]{Because the columns of $A$ are linearly independent.}
  	  		                      \choice{Because $A\vec{x} = \vec{b}$ is has exactly one solution.}
  	  		                      \end{multipleChoice}
  	  		                      
  	  		                      	
  	  		                      \end{question}
