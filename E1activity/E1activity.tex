  	\begin{question}
  		Which of the following is inconsistent?
  		
  		
  		\begin{multipleChoice}
  			\choice{A system of linear equations with infinitely many solutions.}
  			\choice[correct]{A system of linear equations with no solutions.}
  			\choice{A system of linear equations with exactly one unique solution.}
  		\end{multipleChoice}
  		
  	\end{question}

  	\begin{question}
  		Write the following linear system as an augmented matrix.
  		
  			
  			$$\left\{
  			\begin{array}{rrrl}
  			3x&+&2y &= -1\\
  			-5x& -&4y &=0\\
  			\end{array} \right.$$
  			
$  		\begin{amatrix}{2} \answer{3} & \answer{2} &\answer{-1}\\ \answer{-5} &\answer{-4} &\answer{0}\end{amatrix}$
  		
  	\end{question}
