	\begin{question}
		Suppose $T$ is a linear transformation defined by $T(\vec{x}) = A\vec{x}$ and $T\left(\begin{bmatrix} x\\ y \end{bmatrix}\right) = A\begin{bmatrix} x\\ y \end{bmatrix} =\begin{bmatrix} 4x+7y \\ 8y \\ 2x-y \end{bmatrix}$. Then the standard matrix $A$ for this linear transformation is:\\
		
		$A = \left[\begin{array}{cc} \answer{4} & \answer{7} \\\answer{0} &\answer{8} \\ \answer{2} &\answer{-1} \end{array}\right]$
	\end{question}

  		\begin{question}
  	A mapping $T: \RR^n \to \RR^m$ is said to be onto $\RR^m$ if each $\vec{b}$ in $\RR^m$ is the image of at least one $\vec{x}$ in $\RR^n$. Decide which of the following transformations is onto.\\
  	
  	$T(\vec{x}) = A\vec{x}$ where $A = \begin{bmatrix} 1&0\\ 0 &1 \end{bmatrix}$\\
  	
  	\begin{multipleChoice}
  	\choice[correct]{onto}
  	\choice{not onto}
  	\end{multipleChoice}
  	
  	  	$T(\vec{x}) = A\vec{x}$ where $A = \begin{bmatrix} 1&0&2\\ 0 &1 &3 \end{bmatrix}$\\
  	  	
  	  	\begin{multipleChoice}
  	  		\choice[correct]{onto}
  	  		\choice{not onto}
  	  	\end{multipleChoice}
  			
  			
  			  	$T(\vec{x}) = A\vec{x}$ where $A = \begin{bmatrix} 2&1\\ 0 &3 \\ 0&0 \end{bmatrix}$\\
  			  	
  			  	\begin{multipleChoice}
  			  		\choice{onto}
  			  		\choice[correct]{not onto}
  			  	\end{multipleChoice}
  			\end{question}
