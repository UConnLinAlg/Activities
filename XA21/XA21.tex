	\begin{question} Determine which of the following products are defined.\\
		
		$AB= \begin{bmatrix} 1&3&4 \end{bmatrix} \begin{bmatrix} 0\\-3\\2 \end{bmatrix} $
		
		\begin{multipleChoice}
			\choice[correct]{defined}
			\choice{undefined}
			\end{multipleChoice}
			
				$AB=  \begin{bmatrix} 0\\-3\\2 \end{bmatrix} \begin{bmatrix} 1&3&4 \end{bmatrix}$
				
				\begin{multipleChoice}
					\choice[correct]{defined}
					\choice{undefined}
				\end{multipleChoice}
			
			$AB= \begin{bmatrix} 1&3\\ 0&4 \end{bmatrix} \begin{bmatrix} 0 &7\\-3&1\\2&0 \end{bmatrix} $
			
			\begin{multipleChoice}
				\choice{defined}
				\choice[correct]{undefined}
			\end{multipleChoice}
